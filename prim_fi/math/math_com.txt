# Syncronize PC to clock to PLC clock if Difference > 2s
# Convert Daily time to seconds and compare it

OBJ: PCClockSync
     CPAR:
   P0 = PLCClkMonth
   P1 = PLCClkDay
   P2 = PLCClkYear
   P3 = PLCClkHour
   P4 = PLCClkMin
   P5 = PLCClkSec
     CFV: IF (P0!=0) && (P1!=0) && (S5 == 3) : Q1; ELSE Q0;
     OPAR:
   P0  = PLCClkMonth
   P1  = PLCClkDay
   P2  = PLCClkYear
   P3  = PLCClkHour
   P4  = PLCClkMin
   P5  = PLCClkSec
   P6  = SysClkMonth
   P7  = SysClkDay
   P8  = SysClkYear
   P9  = SysClkHour
   P10 = SysClkMin
   P11 = SysClkSec
   P12 = CLK_TSET
     # If (10 < DailyPLCTimeInSec < (86400-10)) and PLCDate!=PCDate then Set PC Time
     OFV0:  P12  = IF ((10 < (P3*60+P4)*60+P5) && ((P3*60+P4)*60+P5 < (86400-10))) && \
                ((P0!=P6) || (P1!=P7) || ((ABS(P2-P8)!=0) && ((ABS(P2-P8)!=100)))) : 1; ELSE ;
     # If (PLCDate==PCDate) and (10 < DailyPLCTimeInSec < (86400-10)) and
     #          (ABS(DailyPLCTimeInSec-DailyPCTimeInSec)>3) then Set PC Time
     OFV1:  P12  = IF ((P0==P6) && (P1==P7) && ((ABS(P2-P8)==0) || ((ABS(P2-P8)==100)))) && \
                ((10 < (P3*60+P4)*60+P5) && ((P3*60+P4)*60+P5 < (86400-10))) && \
                ( ABS( ((P3*60+P4)*60+P5) - ((P9*60+P10)*60+P11) ) > 2) : 1; ELSE ;

OBJ: MBPTaskStop
    CPAR:
        P0 = CLK_DIFF
    CFV: IF (P0<-30 || P0>300) : Q1;ELSE Q0;
    OPAR:
       P0= Task_Restart01
       P1= MBPLUS_SRV_LNCH
       P2= MBPLUS_SRV_STAT
    # If any MB+ is running must be stopped when the clock is
    # changed backwards more than 30 seconds or forward more than 5 minutes
    OFV0: P1 = IF P2 : 0;ELSE;
    OFV1: P0 = IF P2 : 1;ELSE;
